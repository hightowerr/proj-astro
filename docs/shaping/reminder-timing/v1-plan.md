## Step 1 — Schema change

In `src/lib/schema.ts`, extend `bookingSettings`:

```typescript
export const bookingSettings = pgTable(
  "booking_settings",
  {
    shopId: uuid("shop_id")
      .primaryKey()
      .references(() => shops.id, { onDelete: "cascade" }),
    slotMinutes: integer("slot_minutes").notNull(),
    timezone: text("timezone").notNull(),
    // NEW:
    reminderTimings: text("reminder_timings")
      .array()
      .notNull()
      .default(sql`ARRAY['24h']::text[]`),
  },
  (table) => [
    check(
      "booking_settings_slot_minutes_valid",
      sql`${table.slotMinutes} in (15, 30, 45, 60, 90, 120)`
    ),
    // NEW constraints:
    check(
      "booking_settings_reminder_timings_max",
      sql`array_length(${table.reminderTimings}, 1) <= 3`
    ),
    check(
      "booking_settings_reminder_timings_min",
      sql`array_length(${table.reminderTimings}, 1) >= 1`
    ),
    check(
      "booking_settings_reminder_timings_valid",
      sql`${table.reminderTimings} <@ ARRAY['10m','1h','2h','4h','24h','48h','1w']::text[]`
    ),
  ]
);
```

## Step 2 — Generate and edit migration

```bash
pnpm db:generate
```

This produces `drizzle/0023_reminder_timings.sql` (or similar). The generated SQL will add the column and constraints but will NOT backfill existing rows. Manually append the backfill statement:

```sql
-- Generated by drizzle-kit:
ALTER TABLE "booking_settings" ADD COLUMN "reminder_timings" text[] NOT NULL DEFAULT ARRAY['24h']::text[];--> statement-breakpoint
ALTER TABLE "booking_settings" ADD CONSTRAINT "booking_settings_reminder_timings_max" CHECK (array_length("reminder_timings", 1) <= 3);--> statement-breakpoint
ALTER TABLE "booking_settings" ADD CONSTRAINT "booking_settings_reminder_timings_min" CHECK (array_length("reminder_timings", 1) >= 1);--> statement-breakpoint
ALTER TABLE "booking_settings" ADD CONSTRAINT "booking_settings_reminder_timings_valid" CHECK ("reminder_timings" <@ ARRAY['10m','1h','2h','4h','24h','48h','1w']::text[]);--> statement-breakpoint
-- Backfill: existing rows already have the column DEFAULT applied,
-- but run explicitly to guarantee no NULLs (belt-and-suspenders):
UPDATE "booking_settings" SET "reminder_timings" = ARRAY['24h']::text[] WHERE "reminder_timings" IS NULL;
```

> Note: because the column has `NOT NULL DEFAULT`, Postgres applies the default to existing rows automatically on `ALTER TABLE ADD COLUMN`. The explicit `UPDATE` is belt-and-suspenders per R13. Verify row count matches before/after.

## Step 4 — Server action

Create `src/app/app/settings/reminders/actions.ts`:

```typescript
"use server";

import { revalidatePath } from "next/cache";
import { eq } from "drizzle-orm";
import { z } from "zod";
import { db } from "@/lib/db";
import { bookingSettings } from "@/lib/schema";
import { requireAuth } from "@/lib/session";
import { getShopByOwnerId } from "@/lib/queries/shops";

const VALID_INTERVALS = ["10m", "1h", "2h", "4h", "24h", "48h", "1w"] as const;

const schema = z
  .array(z.enum(VALID_INTERVALS))
  .min(1, "Select at least one reminder interval")
  .max(3, "Maximum 3 reminder intervals allowed");

export async function updateReminderTimings(timings: unknown) {
  const session = await requireAuth();
  const shop = await getShopByOwnerId(session.user.id);

  if (!shop) {
    throw new Error("Shop not found");
  }

  const parsed = schema.parse(timings);

  await db
    .update(bookingSettings)
    .set({ reminderTimings: parsed })
    .where(eq(bookingSettings.shopId, shop.id));

  revalidatePath("/app/settings/reminders");
}
```

## Step 6 — Page component

Create `src/app/app/settings/reminders/page.tsx`:

```typescript
import { db } from "@/lib/db";
import { getShopByOwnerId } from "@/lib/queries/shops";
import { requireAuth } from "@/lib/session";
import { ReminderTimingsForm } from "@/components/settings/reminder-timings-form";

type Interval = "10m" | "1h" | "2h" | "4h" | "24h" | "48h" | "1w";

export default async function ReminderSettingsPage() {
  const session = await requireAuth();
  const shop = await getShopByOwnerId(session.user.id);

  if (!shop) {
    return (
      <div className="container mx-auto px-4 py-10">
        <h1 className="text-3xl font-semibold">Reminders</h1>
        <p className="text-sm text-muted-foreground">
          Create your shop to configure reminder settings.
        </p>
      </div>
    );
  }

  const settings = await db.query.bookingSettings.findFirst({
    where: (table, { eq }) => eq(table.shopId, shop.id),
  });

  const initialTimings = (settings?.reminderTimings as Interval[]) ?? ["24h"];

  return (
    <div className="container mx-auto space-y-8 px-4 py-10">
      <header className="space-y-2">
        <h1 className="text-3xl font-semibold">Reminders</h1>
        <p className="text-sm text-muted-foreground">
          Configure when automated reminders send before appointments.
        </p>
      </header>
      <div className="rounded-lg border p-6">
        <ReminderTimingsForm initialTimings={initialTimings} />
      </div>
    </div>
  );
}
```

## Testing

### 8.1 — Migration integrity

```bash
pnpm db:studio
```

Open `booking_settings` table:
- Every existing row has `reminder_timings = {24h}` (not NULL, not empty)
- Column type is `text[]`
- No rows violate the constraints

### 8.2 — Page loads correctly

Ask user to start dev server, then visit `/app/settings/reminders`.

**Expect:**
- Page renders without error
- `24h` preset is pre-selected (slot dots show `1/3`)
- All 7 presets visible with their labels and persona tags
- No "Max 3 reached" banner (only 1 selected)

### 8.3 — Selection behaviour

- Click `10m` → `2/3` dots, `10m` row shows teal check
- Click `2h` → `3/3` dots, coral "Max 3 reached" banner appears
- Try clicking any unselected preset → nothing happens (row is dimmed)
- Deselect `10m` → `2/3` dots, banner disappears, all presets re-enabled
- Save button is active (form is dirty)

### 8.4 — Save and persist

- Select `['10m', '2h', '24h']` → click Save
- "Saving…" text appears briefly, then "Saved" confirmation
- Hard reload page → `['10m', '2h', '24h']` still selected
- Verify in DB: `SELECT reminder_timings FROM booking_settings WHERE shop_id = '<id>'` returns `{10m,2h,24h}`

### 8.5 — Constraint enforcement (schema-level)

Run directly against the DB to confirm constraints hold:

```sql
-- Should fail (4 intervals):
UPDATE booking_settings SET reminder_timings = ARRAY['10m','1h','2h','4h']
WHERE shop_id = '<your-shop-id>';
-- Expected: ERROR: new row violates check constraint "booking_settings_reminder_timings_max"

-- Should fail (invalid interval):
UPDATE booking_settings SET reminder_timings = ARRAY['15min']
WHERE shop_id = '<your-shop-id>';
-- Expected: ERROR: new row violates check constraint "booking_settings_reminder_timings_valid"

-- Should fail (empty):
UPDATE booking_settings SET reminder_timings = ARRAY[]::text[]
WHERE shop_id = '<your-shop-id>';
-- Expected: ERROR: new row violates check constraint "booking_settings_reminder_timings_min"
```

### 8.6 — Save button disabled states

- Fresh load (no changes) → Save is disabled
- Change a selection → Save is enabled
- Click Save → while pending, Save shows "Saving…" and is disabled
- After save, no further changes → Save is disabled again, "Saved" is visible
- Make another change after saving → "Saved" disappears, Save re-enables

### 8.7 — Empty-selection guard

- Deselect all intervals (only possible if starting from 1 selected) → Save stays disabled
- "Select at least one interval" hint text appears

### 8.8 — Accessibility

- Tab through presets: each row receives focus
- Space/Enter toggles selection on focused row
- `aria-pressed` reflects selection state (verify with browser DevTools)
